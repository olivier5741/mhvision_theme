<script>
$(function(){
        $.getScript("https://raw.github.com/swfobject/swfobject/master/swfobject/swfobject.js", function()
{
    var params = {};
    params.wmode = "transparent";
    swfobject.embedSWF("/animation/loin.swf", "loin", "497", "253", "9.0.0", {}, {}, params);

    $.getScript("https://raw.github.com/morr/jquery.appear/master/jquery.appear.js", function()
	{
                var counter = 0;
		$('div#loin-container').appear();
		$('div#loin-container').on("appear",function() {
                        if(counter == 0){
      			   setTimeout(function() {swfobject.getObjectById('loin').Play();}, 2500);
                        }
                        counter ++;
    		});

		$( "div#loin-container" ).click(function() {
			swfobject.getObjectById('loin').Play();
		});
		

	});
    
});
    });
</script>

<div id="loin-container">
    <div id="loin">
        
    </div>
</div>
